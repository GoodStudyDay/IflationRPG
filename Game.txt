      private function loadEND() : void
      {
         var appOff:* = function(param1:Event):void
         {
            stage.frameRate = 0;
            sebgm.appOff();
         };
         var appOn:* = function(param1:Event):void
         {
            if(battleStart)
            {
               if(gamesavedata.speedNum == 0)
               {
                  stage.frameRate = 30;
               }
               else if(gamesavedata.speedNum == 1)
               {
                  stage.frameRate = 45;
               }
               else if(gamesavedata.speedNum == 2)
               {
                  stage.frameRate = 60;
               }
               else if(gamesavedata.speedNum == 3)
               {
                  stage.frameRate = 90;
               }
               else if(gamesavedata.speedNum == 4)
               {
                  stage.frameRate = 120;
               }
            }
            else
            {
               stage.frameRate = 60;
            }
            sebgm.sebgmFunc();
         };
         loadtimef(7);
         stage.addEventListener("mouseDown",MMD);
         stage.addEventListener("mouseUp",MMU);
         stageMap_title_tran = new StageMap_Title_Transitions(this);
         title = new Title(this);
         title.start();
         gamesavedata.playTimes++;
         gamesavedata.sodataFunc(1);
         stage.addEventListener("deactivate",appOff);
         stage.addEventListener("activate",appOn);
         hikitugi = new Hikitugi(this);
         main.contentLoadComp();
      }
      
      public function NetLoadEnd() : void
      {
         if(!isNetLoadInit)

            new DelayFunc(3,main.admob.init);
            disqApplicationRater.Init();
         }
      }
      
      private function loadEEF(param1:Event) : void
      {
         if(loadEEFi == 3)
         {
            load1();
         }
         else if(loadEEFi == 6)
         {
            load2();
         }
         else if(loadEEFi == 9)
         {
            load3();
         }
         else if(loadEEFi == 12)
         {
            load4();
         }
         else if(loadEEFi == 15)
         {
            load5();
         }
         else if(loadEEFi == 18)
         {
            load6();
         }
         else if(loadEEFi == 27)
         {
            removeEventListener("enterFrame",loadEEF);
            loadEND();
         }
         loadEEFi = loadEEFi + 1;
      }
