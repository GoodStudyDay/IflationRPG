      private function EEF(param1:Event) : void
      {
         var _loc9_:int = 0;
         var _loc5_:Number = NaN;
         var _loc8_:int = 0;
         var _loc4_:int = 0;
         var _loc3_:int = 0;
         if(listsV == null)
         {
            return;
         }
         if(pouseEEF)
         {
            return;
         }
         var _loc6_:Number = soubiMenu.mouseX;
         var _loc2_:Number = soubiMenu.mouseY;
         if(mousedown)
         {
            if(tmousedownY == -1)
            {
               tmousedownY = _loc2_;
               tlistSpy = listsSp.y;
               tlisti = -1;
               listdv = 0;
            }
            else if(_loc2_ - tmousedownY > -idousaSikiiNum && _loc2_ - tmousedownY < idousaSikiiNum)
            {
               if(tlisti == -1)
               {
                  var _loc7_:int = int(listsV.length);
                  _loc5_ = Scr.ContentScale;
                  _loc8_ = _loc6_ * _loc5_;
                  _loc4_ = _loc2_ * _loc5_ + ga.co.y;
                  if(Ad32050_v153.admobSmatBigOn)
                  {
                     _loc4_ += 40 * _loc5_;
                  }
                  _loc9_ = 0;
                  while(_loc9_ < _loc7_)
                  {
                     if(listsV[_loc9_].hitTestPoint(_loc8_,_loc4_))
                     {
                        if(listsV[_loc9_].syu != 9)
                        {
                           tlisti = _loc9_;
                           if(listsV[tlisti].b != null)
                           {
                              listsV[tlisti].b.transform.colorTransform = GameConst.ColorDown;
                           }
                           break;
                        }
                     }
                     _loc9_++;
                  }
               }
            }
            else
            {
               if(tlisti != -1)
               {
                  ga.mouseChildren = ga.mouseEnabled = true;
                  if(listsV[tlisti].b != null)
                  {
                     listsV[tlisti].b.transform.colorTransform = GameConst.ColorT;
                  }
                  tlisti = -1;
               }
               listdv = (tlistSpy + (_loc2_ - tmousedownY) - listsSp.y) * 0.9;
               if(listdv > listHeight * 0.9)
               {
                  listdv = listHeight * 0.9;
               }
               else if(listdv < -listHeight * 0.9)
               {
                  listdv = -listHeight * 0.9;
               }
               listsSp.y += listdv;
            }
         }
         else if(tlisti != -1)
         {
            ga.sebgm.okstart();
            listSoubiKakuninOpen(tlisti);
            if(listsV[tlisti].b != null)
            {
               listsV[tlisti].b.transform.colorTransform = GameConst.ColorT;
            }
            tlisti = -1;
         }
         else
         {
            if(listdvSpeedUp == true)
            {
               listdvSpeedUp = false;
               listdv *= 10;
            }
            listsSp.y += listdv;
            listdv *= 0.8;
            if(listdv < 1 && listdv > -1)
            {
               listdv = 0;
            }
            tmousedownY = -1;
         }
         if(listsSp.y > 150)
         {
            listsSp.y = 150;
         }
         else if(listsSp.y < -listAllHeight + Const.AppH - 120)
         {
            if(-listAllHeight + Const.AppH - 120 < 150)
            {
               listsSp.y = -listAllHeight + Const.AppH - 120;
            }
            else
            {
               listsSp.y = 150;
            }
         }
         if(zyouimode == 0)
         {
            _loc3_ = -(listsSp.y - 100) / listHeight;
            if(_loc3_ - 1 >= 0)
            {
               listsV[_loc3_ - 1].visible = true;
            }
            if(_loc3_ >= 0)
            {
               listsV[_loc3_].visible = true;
            }
            _loc3_ = -(listsSp.y - Const.AppH) / listHeight;
            if(_loc3_ < listLength)
            {
               listsV[_loc3_].visible = true;
            }
            if(_loc3_ + 1 < listLength)
            {
               listsV[_loc3_ + 1].visible = true;
            }
         }
         if(tg != ga.gdata.g)
         {
            tg -= (tg - ga.gdata.g) * 0.1 + 11;
            if(ga.gdata.g > tg)
            {
               tg = ga.gdata.g;
            }
            menusoubilists.gtxt.text = tg + "G";
         }
      }

      {

         {
            soubiMenu.mainUpdataBUKI();
            listUpdataBUKIFunc();
         }
         else if(listSentakuSlotNum == 1)
         {
            soubiMenu.mainUpdataBOUGU();
            listUpdataBOUGUFunc();
         }
         else if(listSentakuSlotNum == 16)
         {
            soubiMenu.mainUpdataMATERIAL(listSentakuSlotNum);
            listUpdataMATERIALFunc();
         }
         else if(listSentakuSlotNum >= 2 && listSentakuSlotNum <= 13)
         {
            soubiMenu.mainUpdataAKUSE(listSentakuSlotNum);
            listUpdataAKUSEFunc();
         }
         else if(listSentakuSlotNum >= 14 && listSentakuSlotNum <= 15)
         {
            var _loc2_:Item = SentakuItem;
            ga.gdata.g -= int(Math.pow(1,itemMana.GetItemTimes_Item(_loc2_)) * _loc2_.kakaku);
            itemMana.AddOneItemKazu_Item(_loc2_,-1,-1,-1,-1);
            soubiMenu.mainUpdataSOUL(listSentakuSlotNum);
            listUpdataSOULFunc();
         }
         ga.gdata.EqStUpdate(false);
         soubiMenu.mainUpdataSutetasu();
         listZyouiVisiButton.mouseEnabled = listToziruButton.mouseEnabled = listSetumeiButton.mouseEnabled = true;
