            else if(_loc10_ <= 100 && _loc10_ > 0)
            {
               dropratetxt.text = "Drop : " + _loc11_.name + " (" + _loc10_.toFixed(2) + "%)";
            }
            else
            {
               dropratetxt.text = "Drop : " + _loc11_.name + " (Max)";
            }
         }
         else
         {
            dropratetxt.text = "Drop : " + "------ (0.00%)";
         }
         if(mode == 1)
         {
            ga.gdata.hitCount = 0;
            ga.gdata.hitCount1 = 0;
            mode = 3;
         }
         else if(mode >= 3 && mode <= 5)
         {
            if(mode == 3)
            {
               myStatusSet2();
               if(whichTurn == 0 && buffCount <= 0)
               {
                  myGetDameSkNumFunc();
               }
               else if(whichTurn == 1)
               {
                  eneGetDameSkNumFunc();
               }
               else if(whichTurn == 0 && buffCount >= 0 && ga.BuffSwitch)

                  myGetDameSkNumFunc();

               if(dametiming < 5 && bitMana.SkMotionM > 5)
               {
                  dametiming = 5;
               }
               mode = 4;
            }
            else if(mode == 4)
            {
               if(renzokukaisu >= 2 && buffCount <= 0)
               {
                  if(int(dametiming / 2) == eefi)
                  {
                     renzokuFunc();
                  }
               }
               if(dametiming == eefi && buffCount <= 0)
               {
                  if(whichTurn == 0)
                  {
                     if(ga.gdata.DamageIncreased != 0)
                     {
                        tdame = Math.floor(tdame * ((100 + ga.gdata.DamageIncreased) / 100));
                     }
                     if(ga.gdata.renzoDamageUP != 0 && renzokukaisu >= 2)
                     {
                        tdame = Math.floor(tdame * ga.gdata.renzoDamageUP);
                     }
                     tdame = Math.floor(tdame * eneDamageReduced);
                     recovery();
                     if(ga.gdata.myhprecovery)
                     {
                        myHpRecovery();
                     }
                     trace("MyATK:",myatk);
                     ClearEnemyHp();
                     trace("EneHP:",EneHP);
                     if(Math.random() < ga.gdata.enemissrate && (ga.stagemap.MapNum >= 109 && ga.stagemap.MapNum <= 118))

                        missFunc(false);

                     ga.gdata.hitCount1++;
                     if(ga.gdata.doubleAttack)
                     {
                        myAttack();
                     }
                     if(EneHP < 0)
                     {
                        EneHP = 0;
                     }
                     if(renzokukaisu >= 2)
                     {
                        _loc2_ += L.GetIRep("[0] 連続！",renzokukaisu);
                     }
                     if(crih)
                     {
                        _loc2_ += L.Get("会心！");
                     }

                     {
                        dataManaWL.battleCostG();
                     }
                     if(tdame > ga.gamesavedata.HighDamage)
                     {
                        ga.gamesavedata.HighDamage = tdame;
                     }
                     hpratetxt.text = "HPrate : " + int(EneHP / EneHPM * 100) + "%";
                  }
                  else if(whichTurn == 1)
                  {
                     myhpm = Math.floor(myhpm * ga.gdata.turnAddMaxHP);
                     if(ga.gdata.DamageReduced != 0)
                     {
                        tdame = Math.floor(tdame * ((100 - ga.gdata.DamageReduced) / 100));
                     }
                     if(Math.random() < ga.gdata.missrate || myTurnEndCount < ga.gdata.missrateOn)
                     {
                        trace("missrate:",ga.gdata.missrate);
                        missFunc(true);
                        _loc2_ = "You miss the enemy attack";
                     }
                     else
                     {
                        DamegeMcFunc(tdame,crih,280,600);
                        _loc7_ = myhp;
                        myhp -= tdame;
                        _loc2_ = L.GetNRep("敵の攻撃 [0]ダメ受",Math.floor(tdame));
                     }
                     if(myhp < 0)
                     {
                        myhp = 0;

                        {
                           if(ga.gdata.SokusiKaihiKakuritu >= 1)
                           {

                              }
                           }

                     }
                     ga.gdata.hitCount++;
                  }
                  battleLogText.addTextBattleLogNouUpdata(_loc2_);
                  enePattern(BossSyu);
                  if(whichTurn == 1 && myhp > 0 && ga.gdata.reflection > 0)
                  {

                        }
                     }

                  }
               }
               if(bitMana.EEFBattleFunc4(whichTurn))
               {
                  mode = 5;
               }
            }
            else if(mode == 5)
            {
               if(eefi >= 2)
               {

                  if(whichTurn == 0 && buffCount <= 0)
                  {
                     myTurnEndCount = myTurnEndCount + 1;
                     if(EneHP > 0)
                     {

                        {
                           if(barMana.tmyhpbar <= 0.5)
                           {
                              _loc3_ += 0.13;
                           }
                           else if(barMana.tmyhpbar <= 0.1)
                           {
                              _loc3_ += 0.11;
                           }
                           else if(barMana.tmyhpbar <= 0.25)
                           {
                              _loc3_ += 0.08;
                           }
                           if(_loc3_ >= 0.8)
                           {
                              _loc3_ = 0.8;
                           }
                        }
                        if(BossSyu == 76)
                        {
                           _loc3_ *= 0.7;
                        }
                        renzokutxt.text = "Combo : " + int(_loc3_ * 100) + "%";
                        if(Math.random() <= _loc3_)
                        {
                           whichTurn = 0;
                           renzokukaisu = renzokukaisu + 1;
                           if(renzokukaisu > ga.gamesavedata.HighCombo)
                           {
                              ga.gamesavedata.HighCombo = renzokukaisu;
                           }
                        }
                        else
                        {
                           eneDouble = Math.random() * 100;
                           eneSkill = Math.random() * 100;
                           whichTurn = 1;
                           renzokukaisu = 1;
                        }
                        mode = 3;
                     }
                     else if(eneHP2 <= 0)
                     {
                        mode = 110;
                        dataManaWL.StartWin(BossSyu);
                     }
                     else
                     {
                        mode = 3;
                     }
                  }
                  else if(myhp > 0)
                  {
                     whichTurn = 0;
                     buffCount = buffCount - 1;
                     if(kaihukuNextTurnOn)
                     {
                        mode = 6;
                        kaihukuNextTurnOn = false;
                     }
                     else
                     {
                        mode = 3;
                     }
                  }
                  else if(ga.gdata.resCount != 0)
                  {
                     mode = 6;
                     resSwitch = 1;
                     whichTurn = 0;
                  }
                  else
                  {
                     mode = 160;
                     dataManaWL.StartLose();
                  }
               }

         }

            }
         }
      }
