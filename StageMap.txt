      private function MapLoadStart(param1:int) : void
      {
         var _loc2_:int = 0;
         var _loc3_:int = 0;
         ga.sebgm.bgmstopf();
         mapLoadEEFSyurui = param1;
         if(mapLoadEEFSyurui == 3)
         {
            mapLoadEEFdankai = 5;
            StageMapLoad.alpha = 1;
         }
         else
         {
            mapLoadEEFdankai = 0;
            StageMapLoad.alpha = 0;
         }
         if(ga.gamesavedata.HakkenmapV[MapNum] == 0)
         {
            AppRaterMana.FirstMapV(MapNum);
         }
         else
         {
            AppRaterMana.NotFirstMapV(MapNum);
         }
         ga.gamesavedata.HakkenmapV[MapNum] = 1;
         if(mapLoadEEFSyurui != 0)
         {
            mukindy = mukindx = 0;
         }
         if(mapLoadEEFSyurui != 1 || mapLoadEEFSyurui != 4)
         {
            if(ga.gdata.BonasDataMapNum[0] != MapNum)
            {
               if(MapNum == ga.gdata.BonasDataMapNum[2])
               {
                  ga.gdata.BonasDataMapNum[2] = ga.gdata.BonasDataMapNum[1];
                  ga.gdata.BonasDataMapNum[1] = ga.gdata.BonasDataMapNum[0];
                  ga.gdata.BonasDataMapNum[0] = MapNum;
                  _loc3_ = 0;
                  while(_loc3_ < 40)
                  {
                     _loc2_ = ga.gdata.StageLvBonasData3[_loc3_];
                     ga.gdata.StageLvBonasData3[_loc3_] = ga.gdata.StageLvBonasData2[_loc3_];
                     ga.gdata.StageLvBonasData2[_loc3_] = ga.gdata.StageLvBonasData1[_loc3_];
                     ga.gdata.StageLvBonasData1[_loc3_] = _loc2_;
                     _loc3_++;
                  }
               }
               else if(MapNum == ga.gdata.BonasDataMapNum[1])
               {
                  ga.gdata.BonasDataMapNum[1] = ga.gdata.BonasDataMapNum[0];
                  ga.gdata.BonasDataMapNum[0] = MapNum;
                  _loc3_ = 0;
                  while(_loc3_ < 40)
                  {
                     _loc2_ = ga.gdata.StageLvBonasData2[_loc3_];
                     ga.gdata.StageLvBonasData2[_loc3_] = ga.gdata.StageLvBonasData1[_loc3_];
                     ga.gdata.StageLvBonasData1[_loc3_] = _loc2_;
                     _loc3_++;
                  }
               }
               else
               {
                  ga.gdata.BonasDataMapNum[2] = ga.gdata.BonasDataMapNum[1];
                  ga.gdata.BonasDataMapNum[1] = ga.gdata.BonasDataMapNum[0];
                  ga.gdata.BonasDataMapNum[0] = MapNum;
                  _loc3_ = 0;
                  while(_loc3_ < 40)
                  {
                     ga.gdata.StageLvBonasData3[_loc3_] = ga.gdata.StageLvBonasData2[_loc3_];
                     ga.gdata.StageLvBonasData2[_loc3_] = ga.gdata.StageLvBonasData1[_loc3_];
                     ga.gdata.StageLvBonasData1[_loc3_] = 0;
                     _loc3_++;
                  }
               }
            }
         }

         removeEventListener("enterFrame",MapEEF);

      }
